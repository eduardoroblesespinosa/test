                            Muaz Khan — 


                            // connection.userid = 'userA-abc';
                            // connection.userid = 'userA-def';
                            // connection.userid = 'userB-abc';
                            // connection.userid = 'userA-def';
                            connection.onstream = function(event) {
                                var firstPartOfRemoteUserId = event.userid.split('-')[0];

                                if (connection.isInitiator === true && event.type === 'remote') {
                                    // master peer relays/forwards to other participants
                                    connection.getAllParticipants().forEach(function(pid) {
                                        if (pid === event.userid) return; // ignore streamer

                                        var firstPartOfPID = pid.split('-')[0];
                                        if (firstPartOfRemoteUserId !== firstPartOfPID) return; // ignore of both users do not have same id formats

                                        var peer = connection.peers[pid].peer;
                                        peer.addStream(event.stream);
                                    });

                                    connection.renegotiate();
                                }
                            };

                            You can even use extra data:


                            // connection.extra.group = 'A';
                            connection.onstream = function(event) {
                                var remoteUserGroup = event.extra.group;

                                if (connection.isInitiator === true && event.type === 'remote') {
                                    // master peer relays/forwards to other participants
                                    connection.getAllParticipants().forEach(function(pid) {
                                        if (pid === event.userid) return; // ignore streamer

                                        if (connection.peers[pid].extra.group !== remoteUserGroup) return; // ignore if group does not matches

                                        var peer = connection.peers[pid].peer;
                                        peer.addStream(event.stream);
                                    });

                                    connection.renegotiate();
                                }
                            };
